from PyQt5.QtWidgets import QApplication, QFileDialog, QWidget, QPushButton


[docs]class Example(QWidget):
    
    def __init__(self):
        super().__init__()
        btn = QPushButton('Choisir un dossier', self)
        self.btn=btn
        self.initUI()
        

[docs]    def initUI(self):
        self.setGeometry(300, 300, 300, 200)
        self.setWindowTitle('Choisir un dossier')
        
        self.btn.clicked.connect(self.showDialog)
        self.btn.move(50, 50)
        
        self.show()

        

[docs]    def showDialog(self):
        options = QFileDialog.Options()
        options |= QFileDialog.ShowDirsOnly
        dossier = QFileDialog.getExistingDirectory(self, "Choisir un dossier", options=options)
        self.btn.setText(dossier)
        print(self.btn.text())

if __name__ == '__main__':
